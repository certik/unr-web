\section*{Quiz 10}

\subsection*{Problem 1}

Find the derivatives of:
$$f(x) = \sqrt x(x-1)$$
$$f(x) = x$$
$$f(x) = x^2$$
$$f(x) = {1\over x^2}$$
$$f(x) = \sqrt[3]{x^2}$$

\section*{Solution to the Problem 1}
$$f'(x) = (\sqrt x x-\sqrt x)' = (x^{3/2}-x^{1/2})'={3\over2}x^{1/2}-{1\over2}x^{-1/2}$$
$$f'(x) = 1$$
$$f'(x) = 2x$$
$$f'(x) = (x^{-2})' = -2x^{-3}$$
$$f'(x) = (x^{2/3})' = {2\over3}x^{-1/3}$$
